\subsection{Эксперименты}
\label{Experiments}

Мы рассматриваем алгоритмы Adam~\citep{kingma2014adam} и OASIS~\citep{jahani2021doubly}, а также их вариации. Их основное отличие заключается в способе вычисления матрицы предобуславливания: в Adam это диагональная матрица, составленная из экспоненциально усреднённых квадратов компонент градиента (правило \eqref{eq:squares}), в OASIS --- стохастическая оценка диагонали гессиана по векторам из распределения Радемахера (правило \eqref{eq:linear}). Три варианта регуляризации для Adam и OASIS приведены в Алгоритме \ref{alg:genAdam} и Алгоритме \ref{alg:OASIS} соответственно; цветовая схема согласована с Алгоритмом \ref{alg:precond}: \textcolor{blue}{синий} --- обычная регуляризация (L2), \textcolor{orange}{оранжевый} --- масштабированное затухание весов (WH), \textcolor{red}{красный} --- затухание весов (W).

\begin{algorithm}[H]
            \caption{Различные способы добавления регуляризации для Adam}
            \label{alg:genAdam}
            \begin{algorithmic}
            \small{
            \Require{$\eta, \beta_1, \beta_2, \epsilon$ --- гиперпараметры, $f$ --- функция потерь, $r$ --- регуляризатор, $w_0$ --- начальная точка}
            \State $m_0 = 0$, $v_0 = 0$, $t = 0$
            \While {$w$ не сойдётся}
            \State $t = t+1$
            \State $g_t = \nabla f(w_{t-1}) + $ \textcolor{blue}{$\nabla r(w_{t-1})$}\hfill \textcolor{blue}{AdamL2}
            \State $m_t = \beta_1 \cdot m_{t-1} + (1 - \beta_1) \cdot g_t$
            \State $v_t = \beta_2 \cdot v_{t-1} + (1 - \beta_2) \cdot g_t^2$
            \State $\hat{m}_t = \frac{m_t}{1-\beta_1^t} +$ \textcolor{orange}{$\nabla r(w_{t-1})$} \hfill \textcolor{orange}{AdamWH}
            \State $\hat{v}_t = \frac{v_t}{1-\beta_2^t}$
            \State $w_t = w_{t-1} - \eta \cdot \frac{\hat{m}_t}{\sqrt{\hat{v}_t} + \epsilon} - $ \textcolor{red}{$\eta \nabla r(w_{t-1})$ } \hfill \textcolor{red}{AdamW}
            \EndWhile
            }
\end{algorithmic}
\end{algorithm}

\begin{algorithm}[H]
\caption{Различные способы добавления регуляризации для OASIS}\label{alg:OASIS}
\begin{algorithmic}
    \Require{$\eta_0, D_0, \beta_2, \alpha$ --- гиперпараметры, $f$ --- функция потерь, $r$ --- регуляризатор, $w_0$ --- начальная точка, $\theta_0 = +\infty$ --- начальное значение для адаптивного шага}
    \State $\hat{D}_0^{ii} = \max \{\alpha, |D_0^{ii}| \}$, $\forall i = \overline{1, d}$
    \State $w_1 = w_0 - \eta_0 \hat{D}_0^{-1} \nabla f(w_0)$

    \For{$k = 1, 2, ...$}
    \State $g_k = \nabla f(w_k) +$ \textcolor{blue}{$\nabla r(w_{k})$}\hfill \textcolor{blue}{OASISL2}
    \State $z_k \sim$ Радемахер, т.е. $z_k^i \in \{-1, 1\}$ равновероятно
    \State $D_k = \beta_2 D_{k-1} + (1-\beta_2) \cdot \mathrm{diag}\left( z_k \odot \nabla^2 \left(f(w_k) + \textcolor{blue}{r(w_k)} \right) z_k \right)$ \hfill \textcolor{blue}{OASISL2}
    \State $\hat{D}_k^{ii} = \max \{\alpha, |D_k^{ii}| \}$, $\forall i = \overline{1, d}$
    \State $\eta_k = \min \left\{ \sqrt{1 + \theta_{k-1}} \cdot \eta_{k-1}; \frac{\|w_k - w_{k-1}\|_{\hat{D}_k}}{2 \|\nabla f(w_k) - \nabla f(w_{k-1}) \|_{\hat{D}_k}^* } \right\}$
    \State \textcolor{blue}{$w_{k+1} = w_k - \eta_k \hat{D}_k^{-1} g_k$} \hfill \textcolor{blue}{OASISL2}
    \State \textcolor{orange}{$w_{k+1} = w_k - \eta_k \hat{D}_k^{-1} \left(g_k + \nabla r(w_k)\right)$} \hfill \textcolor{orange}{OASISWH}
    \State \textcolor{red}{$w_{k+1} = w_k - \eta_k \hat{D}_k^{-1} g_k - \eta_k \nabla r(w_k)$} \hfill \textcolor{red}{OASISW}
    \State $\theta_k = \frac{\eta_k}{\eta_{k-1}}$
    \EndFor
\end{algorithmic}
\end{algorithm}

Отметим, что в синем варианте (OASISL2) регуляризатор участвует и в градиенте $g_k$, и в оценке гессиана (то есть в матрице $D_k$) --- это в точности применение базового метода к задаче \eqref{F_big}. В оранжевом и красном вариантах матрица $D_k$ строится только по $f$, а $g_k = \nabla f(w_k)$.

\subsubsection{Методика экспериментов}

Все эксперименты выполнены на графических ускорителях NVIDIA GeForce RTX 3090. На задачах малой размерности (данный и следующие четыре подраздела) вычисления проводятся в двойной точности (float64) и, если не оговорено противное, с полным градиентом; эксперименты с нейронными сетями (три заключительных подраздела) используют собственные протоколы, описанные по месту. Матрица предобуславливания срезается с двух сторон: $\hat{D}_t = \mathrm{clip}(|D_t|, \alpha, \Gamma)$ с $\alpha = 10^{-8}$, $\Gamma = 10^3$ --- двусторонняя срезка обеспечивает контролируемость констант из предположения \ref{ass:preconditioned}. Критерий $\|\nabla \widetilde{F}_t\|^2 = \|\nabla f(w_t) + \hat{D}_t \nabla r(w_t)\|^2$ всюду вычисляется с той же матрицей $\hat{D}_t$, которая использована в шаге метода на данной итерации.

Задачи: (а) диагональная квадратичная $f(w) = \sum_i \left(\frac{a_i}{2} (w^i)^2 - b_i w^i\right)$, $d = 100$, спектр лог-равномерный, $\kappa = a_{\max}/a_{\min} \in \{10, 10^3\}$ --- для неё решения обеих задач известны аналитически: $w^{*,i} = b_i/(a_i + \lambda)$ и $\widetilde{w}^{*,i} = b_i/(a_i + \lambda d^i)$; (б) плотная квадратичная с тем же спектром; (в) логистическая регрессия с регуляризацией Тихонова на наборе данных \texttt{mushrooms} из LIBSVM ($8124$ объектов, $d = 112$, $\lambda = 0.04$); (г) для стохастических экспериментов --- гладкая невыпуклая задача классификации $f(w) = \frac{1}{n}\sum_j (1 - \tanh(y_j x_j^T w))$ с ограниченным сепарабельным регуляризатором $r(w) = \lambda \sum_i \frac{(w^i)^2}{1 + (w^i)^2}$ (удовлетворяет предположению \ref{ass:regbound}; используется для теоремы \ref{theor:3}), а также диагональная квадратичная задача с аддитивным гауссовским шумом с нулевым средним и контролируемой дисперсией $\sigma^2$ (проверка шумовых членов; для теоремы \ref{theor:4} --- с регуляризацией Тихонова). Точные решения $w^*$ и $\widetilde{w}^*$ вычисляются методом Ньютона до нормы градиента $10^{-12}$; шаг $\eta$ подбирается по сетке отдельно для каждого метода; каждая кривая --- медиана по 5 запускам из случайных начальных точек.

Валидация стенда: для диагональной квадратичной задачи траектория AdamW после стабилизации матрицы $\hat D$ совпадает с аналитическим решением $\widetilde{w}^*(\hat D)$ с относительной точностью $8 \cdot 10^{-17}$ (машинная точность), при этом $\|\nabla \widetilde{F}\|^2 \approx 10^{-30}$, а $\|\nabla F\|^2 = 0.11$ --- отношение критериев составляет $10^{29}$.

Рис. \ref{fig:main_adam} и \ref{fig:main_oasis} демонстрируют главное утверждение работы. У методов с затуханием весов (AdamW, OASISW) норма градиента исходной функции $F$ не убывает --- она стабилизируется на ненулевом уровне порядка $\|(I - D_t)\nabla r(w_t)\|$, что согласуется с леммой \ref{lemma:lowerbondF}, --- тогда как квадрат нормы градиента изменённой функции $\widetilde{F}_t$ монотонно убывает на 10--16 порядков, в согласии с теоремами \ref{theor:1} и \ref{theor:2}. У метода с обычной регуляризацией (L2) картина зеркальная: убывает $\|\nabla F\|^2$ (до уровня, задаваемого дискретизацией шага, см. ниже), а $\|\nabla \widetilde{F}_t\|^2$ выходит на плато. Промежуточный метод (WH) обсуждается отдельно ниже. Такое же поведение наблюдается на обеих квадратичных задачах для OASIS-семейства; для Adam-семейства на плохо обусловленных квадратичных задачах ($\kappa = 10^3$) контраст критериев при умеренных значениях шага частично маскируется плато, обусловленным дискретизацией (осцилляции Adam с постоянным шагом), и предельная точка проверяется напрямую --- сравнением с аналитическим $\widetilde{w}^*$, см. выше.

Для 27 конфигураций (3 задачи $\times$ 9 значений $\lambda \in [10^{-4}, 1]$) вычислены точные решения $w^*$ и $\widetilde{w}^*$ и обе части неравенства леммы \ref{lemma:lowerbondF}. Рис. \ref{fig:dinf} иллюстрирует интерпретацию из раздела о решении: элементы $d^i$ стационарной матрицы предобуславливания сильно коррелируют с частотой соответствующих признаков (корреляция Спирмена $0.876$, $p \approx 10^{-36}$), то есть эффективный покоординатный штраф $\lambda d^i$ мал именно там, где градиенты малы. Как следствие, на редких признаках решение AdamW имеет средний модуль веса в $\approx 50$ раз больший, чем решение AdamL2 ($3.47$ против $0.07$): затухание весов практически не штрафует редко активируемые координаты.

Отдельно проверено поведение самой матрицы $D_t$ (рис. \ref{fig:dstab}): поэлементного предела $\lim_t D_t$ в строгом смысле не наблюдается --- в детерминированном полноградиентном режиме $v_t \to 0$ вблизи стационарной точки и $\hat D_t$ медленно дрейфует к нижней срезке, а в стохастическом режиме $\hat D_t$ выходит на уровень, определяемый шумом градиента, и флуктуирует вокруг него. Это оправдывает формулировку леммы \ref{lemma:lowerbondF} для фиксированного шага $t$ (без предположения о существовании предела).

На квадратичной задаче с точно известными константами $\mu_f, L_f$ величина $R_t^2 = \|w_t - \widetilde{w}^*_t\|^2_{D_t}$ (где $\widetilde{w}^*_t$ пересчитывается по текущей $D_t$ прямым решением линейной системы) убывает линейно на 25--28 порядков до машинной точности (рис. \ref{fig:t2_rate}). Для проверки стохастических теорем используется аддитивный несмещённый шум с контролируемой дисперсией $\sigma^2$ и <<запаздывающий>> предобуславливатель (предположение \ref{ass:expectations}). Шумовой член теоремы \ref{theor:3} проверен на предобуславливателе OASIS-типа, который не зависит от шума градиента (оценка гессиана): измеренное плато $\mathbb{E}\|\nabla\widetilde{F}_t\|^2$ масштабируется по $\eta$ с наклоном $1.00$ и по $\sigma^2$ с наклоном $1.00$ в двойных логарифмических координатах (рис. \ref{fig:t3_noise}) --- в точном согласии с членом $\Gamma\widetilde{L}\eta\sigma^2/\alpha^2$. Для теоремы \ref{theor:4} стационарный уровень $\mathbb{E}\|w_t - \widetilde{w}^*_t\|^2_{D_t}$ растёт с наклоном $1.03$ по $\eta$, масштабируется в точности как $\sigma^2$ и всюду лежит ниже теоретической границы (рис. \ref{fig:t4_ball}).

Два замечания об области применимости оценок. Во-первых, для Adam-типа предобуславливателя ($D$ строится по квадратам стохастических градиентов) зависимость плато от $\eta$ на практике слабее линейной: $\hat D_t$ сам подстраивается под уровень шума и частично <<самонормализует>> шаг; граница теоремы \ref{theor:3} при этом выполняется, но по $\eta$ консервативна. Во-вторых, дрейфовые члены с $\rho_\beta$ (теоремы \ref{theor:1}--\ref{theor:4}) --- оценки худшего случая: сама лемма \ref{lemma:Dt} подтверждается напрямую (измеренное $\max_t\|\hat D_{t+1} - \hat D_t\|_\infty$ масштабируется как $1-\beta$ и никогда не нарушает границу), однако в детерминированном полноградиентном режиме Adam-типа $v_t \to 0$ у стационарной точки, матрица останавливается на срезке и дрейфовое плато вовсе не проявляется, а в стохастическом режиме уровень плато определяется шумовым членом.

\subsubsection{Свойства масштабированного затухания весов и мосты к практическому Adam}

Эксперименты позволяют охарактеризовать предложенный промежуточный способ регуляризации. Во-первых, его стационарная точка --- решение $w^*$ \emph{исходной} задачи \eqref{F_big}: условие стационарности (неподвижной точки) $\hat D_t^{-1}(\nabla f(w) + \nabla r(w)) = 0$ эквивалентно $\nabla F(w) = 0$. На \texttt{mushrooms} при индивидуально подобранных шаге и срезке AdamWH и AdamL2 сходятся к $w^*$ с точностью $10^{-15}$, тогда как AdamW уходит от $w^*$ на расстояние $6$--$10$. На рис. \ref{fig:main_adam} (слева) у AdamWH $\|\nabla F\|^2 \to 10^{-31}$; плато AdamL2 на том же рисунке ($\sim 10^{-4}$) обусловлено дискретизацией при выбранном для него шаге и не является свойством предельной точки. При этом, в отличие от L2, регуляризатор не искажает матрицу предобуславливания.

Во-вторых, у метода есть ограничение устойчивости: регуляризатор проходит через $\hat D_t^{-1}$, и на <<мёртвых>> координатах (где градиент мал и $\hat D_t$ упирается в нижнюю срезку $\alpha$) эффективный шаг по регуляризатору возрастает до $\eta\lambda/\alpha$. Это даёт простое необходимое условие устойчивости $\eta\lambda/\alpha < 2$, которое подтверждается экспериментом (таблица \ref{tab:wh}): во всех пяти конфигурациях, включая пограничную $\eta\lambda/\alpha = 1$, наблюдение совпадает с предсказанием правила. Практический вывод: WH следует использовать с умеренной нижней срезкой (или $\epsilon$), $\alpha \gtrsim \eta\lambda$.

\subsubsection{Нейронные сети: ResNet-18 на CIFAR-10}

Перейдём к нейронным сетям и проверим, что найденные эффекты сохраняются при их обучении. Сеть ResNet-18~\citep{he2015deep} (вариант для CIFAR: свёртка $3{\times}3$ на входе, без начального пулинга) обучается на CIFAR-10~\citep{krizhevsky2009learning}: 50 эпох, размер батча 128, стандартные аугментации, косинусное расписание шага с разогревом 5 эпох, $(\beta_1, \beta_2, \epsilon) = (0.9, 0.999, 10^{-8})$, смешанная точность. Сравниваются AdamL2, AdamW и AdamWH на полной сетке $\eta \in \{10^{-4}, 3\cdot10^{-4}, 10^{-3}, 3\cdot10^{-3}, 10^{-2}\} \times \lambda \in \{10^{-5}, 10^{-4}, 10^{-3}, 10^{-2}, 10^{-1}, 1\}$ --- по 30 пар $(\eta, \lambda)$ на метод, всего 90 прогонов. Критерии $\|\nabla F\|^2$ и $\|\nabla \widetilde{F}_t\|^2$ измеряются на фиксированном пробном батче из 2048 обучающих примеров без аугментаций, причём в качестве $\hat D_t$ берётся ровно та матрица $\sqrt{\hat v_t} + \epsilon$, что используется в шаге оптимизатора.

Результаты (рис. \ref{fig:b1_heatmap}) воспроизводят и уточняют известную картину~\citep{loshchilov2017decoupled}. Лучшую точность даёт AdamW: по трём запускам в оптимуме ($\eta = 10^{-2}$, $\lambda = 0.1$) --- $93.9 \pm 0.3\%$ (максимум $94.2\%$) против $93.8 \pm 0.1\%$ у AdamWH и $93.5 \pm 0.1\%$ у AdamL2; преимущество AdamW над AdamL2 превышает разброс между запусками. Главное различие --- в форме областей оптимума. У AdamW качество практически нечувствительно к $\lambda$ на пяти порядках: даже при $\lambda = 1$ средняя по $\eta$ точность составляет $89.8\%$ --- это и есть <<развязка>> $\lambda$ от $\eta$. У AdamL2 область оптимума диагональна: с ростом $\lambda$ оптимальный шаг сползает вниз (регуляризатор попадает в $D_t$ и меняет эффективный масштаб шага), а при $\lambda = 1$ средняя точность падает до $38.9\%$; в точке $(\eta, \lambda) = (10^{-2}, 0.1)$ прямое сравнение даёт $94.2\%$ у AdamW против $19.7\%$ у AdamL2. У AdamWH собственной развязки нет: при малых $\lambda$ он ведёт себя как AdamW и AdamL2, но правая часть карты обрушивается вдоль линии $\eta\lambda \approx \mathrm{const}$ (при $\lambda = 1$ средняя точность $10.8\%$ --- уровень случайного угадывания) --- это то же ограничение устойчивости, что и в таблице \ref{tab:wh}, --- роль нижней срезки $\alpha$ на практике играет реализовавшийся масштаб знаменателя $\sqrt{\hat v_t} + \epsilon$.

Основной эффект теории воспроизводится на уровне всей сети, а не отдельного слоя (рис. \ref{fig:net_criteria}): к концу обучения лучшей конфигурации AdamW квадрат нормы стохастического градиента $\|\nabla f\|^2$ на пробном батче падает на четыре порядка (с $3.6\cdot10^2$ до $1.9\cdot10^{-2}$), норма $\|\nabla \widetilde{F}_t\|^2$ убывает вместе с ним, тогда как $\|\nabla F\|^2 = \|\nabla f + \lambda w\|^2$ выходит на плато $\sim 6\cdot10^2$ --- расхождение критериев к концу обучения составляет от $9\cdot10^3$ до $3.3\cdot10^4$ раз по трём запускам. Послойная статистика $\hat D_t$ подтверждает механизм адаптивной регуляризации и на ResNet-18: медианные значения $\hat d^i$ различаются между группами слоёв на три порядка (от $5\cdot10^{-6}$ в глубоких свёрточных блоках до $9\cdot10^{-3}$ в первом батч-нормировочном слое), то есть эффективный покоординатный штраф $\lambda \hat d^i$ сильно неоднороден по сети --- в полном согласии с интерпретацией из раздела о решении. Отдельная серия запусков с постоянным шагом и $\beta_2 \in \{0.9, 0.99, 0.999, 0.9999\}$ подтверждает сделанное выше замечание об области применимости дрейфовых оценок: уровень плато $\|\nabla \widetilde{F}_t\|^2$ на сети определяется шумом мини-батчей и практически не зависит от $\beta_2$ (наклон $0.15$ в двойных логарифмических координатах), оставаясь при этом на полтора--два порядка ниже $\|\nabla F\|^2$.

\subsubsection{Масштабирование: предобучение GPT-2 и дообучение на GLUE}

Наконец, проверим, что найденные эффекты сохраняются на масштабе языковых моделей. Модель GPT-2 (124M параметров: 12 слоёв, размерность 768, контекст 1024) предобучается на корпусе FineWeb-Edu ($3.2$ млрд токенов, токенизация GPT-2 BPE): батч $0.245$M токенов, косинусное расписание с разогревом, $(\beta_1, \beta_2, \epsilon) = (0.9, 0.95, 10^{-8})$, смешанная точность. Гиперпараметры $(\eta, \lambda)$ подбирались для каждого метода отдельно на пилотной сетке (13 запусков по $100$M токенов); основные запуски --- $1.5$ млрд токенов, по два запуска на метод из разных начальных точек.

Пилотная сетка воспроизводит выводы предыдущих разделов на новом масштабе. AdamW практически нечувствителен к гиперпараметрам: изменение $\lambda$ в $10$ раз (от $0.01$ до $0.1$) меняет валидационную ошибку лишь на $0.0002$ ната ($5.8346$ против $5.8348$), тогда как AdamL2 заметно хуже и сильно $\lambda$-чувствителен ($6.86$ при $\lambda = 0.01$ и $8.06$ при $\lambda = 0.1$): регуляризатор, попадающий в $D_t$, искажает эффективный шаг. AdamWH расходится во всех пяти испытанных конфигурациях, включая $\lambda = 10^{-3}$: у языковой модели уже на инициализации есть параметры с почти нулевыми градиентами (строки эмбеддингов редких токенов), для которых покоординатный порог устойчивости нарушен на несколько порядков. Правило устойчивости работает и в <<лечащую>> сторону: поднятие $\epsilon$ до $10^{-4}$--$10^{-3} \gg \eta\lambda$ полностью устраняет расходимость (ни одного nan), --- однако вылеченный метод неконкурентоспособен: при том же бюджете токенов он останавливается на валидационной ошибке $7.22$ ($\epsilon = 10^{-4}$) и $7.85$ ($\epsilon = 10^{-3}$) против $5.83$ у AdamW --- большая аддитивная константа в знаменателе сплющивает предобуславливатель и уничтожает саму адаптивность. Вместе с результатами на \texttt{mushrooms} (таблица \ref{tab:wh}) и CIFAR-10 (рис. \ref{fig:b1_heatmap}) это очерчивает область применимости масштабированного затухания весов: рабочее окно $\lambda$ сужается вместе с типичным масштабом градиентов, и на предобучении языковых моделей рабочей точки не остаётся --- либо расходимость (малый $\epsilon$), либо деградация (большой $\epsilon$).

Все три ветки ведут себя согласованно: валидационная ошибка убывает одинаково ($3.516 \to 3.506 \pm 0.001$), а отношение критериев $\|\nabla F\|^2/\|\nabla \widetilde{F}_t\|^2$ совпадает до второго знака ($3.33$--$3.36$), --- наблюдаемые эффекты являются свойством метода, а не конкретной случайной траектории.

Отметим уточнение к интерпретации послойной адаптивности в режиме дообучения: медианные $\hat d^i$ у \emph{верхних} слоёв энкодера примерно в $4$ раза меньше, чем у нижних (у верхних слоёв градиенты быстрее затухают по мере подстройки под задачу), поэтому эффективный штраф $\lambda \hat d^i$ щадит именно верхние слои. Направление адаптации задаётся распределением градиентов конкретной модели и режима обучения, а не фиксировано архитектурой.
